CREATE SCHEMA `mydb` DEFAULT CHARACTER SET utf8 ;

CREATE TABLE `mydb`.`mvc_board` (
  `bNumber` INT NOT NULL AUTO_INCREMENT,
  `bAuthor` VARCHAR(20) NULL,
  `bTitle` VARCHAR(100) NULL,
  `bContent` VARCHAR(300) NULL,
  `bDate` DATE NULL,
  `bHit` INT NULL,
  `bGroup` INT NULL,
  `bStep` INT NULL,
  `bIndent` INT NULL,
  PRIMARY KEY (`bNumber`));
